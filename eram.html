<!DOCTYPE html>

<html>

<head>

    <title>Create New User</title>

    <script src="https://cdn.jsdelivr.net/gh/davidshimjs/qrcodejs/qrcode.min.js"></script>

</head>

<body>


<h1>Create New User</h1>


<form id="userForm">

    <label for="username">Username:</label><br>

    <input type="text" id="username" name="username"><br><br>

    <input type="submit" value="Submit">

</form> 


<h2>List of Users:</h2>


<table id="userTable">

  <tr>

    <th>Name</th>

    <th>QR Code</th> 

    <th>Access Count</th>

  </tr>

</table>


<script>

  let accessCounts = {}


  document.getElementById("userForm").addEventListener("submit", function(event){

      event.preventDefault()

      let username = document.getElementById("username").value

      let userTable = document.getElementById("userTable")

      let newRow = userTable.insertRow(-1)

      let nameCell = newRow.insertCell(0)

      let qrCodeCell = newRow.insertCell(1)

      let accessCountCell = newRow.insertCell(2)


      nameCell.innerHTML = `<a href="/${username}" target="_blank">${username}</a>`

      

      accessCounts[username] = 0

      accessCountCell.innerHTML = accessCounts[username]


      let qrCodeElement = document.createElement('div')

      qrCodeElement.id = `qrcode-${username}`

      

      qrCodeCell.appendChild(qrCodeElement)


      new QRCode(qrCodeElement, {

        text: `https://bing.com/${username}`,

        width: 64,

        height: 64,

        correctLevel : QRCode.CorrectLevel.H

      })


      document.getElementById("username").value = ""

  })


  window.addEventListener('blur', function() {

    if (document.activeElement.href) {

        const usernameMatch = document.activeElement.href.match(/\/([^\/]+)$/)

        if (usernameMatch) {

            const username = usernameMatch[1]

            if (accessCounts.hasOwnProperty(username)) {

                accessCounts[username] += 1

                const rowToUpdateIndex =

                    Array.from(document.querySelectorAll('#userTable tr'))

                        .findIndex(row => row.cells[0].textContent === username)

                if (rowToUpdateIndex !== -1) {

                    document.querySelector(`#userTable tr:nth-child(${rowToUpdateIndex + 1}) td:nth-child(3)`).textContent =

                        accessCounts[username]

                }

            }

        }

    }

})

  

</script>


</body>

</html>
